 bool palindrom (char s1[], int length)
 {
	string s;
	for (int i = 0; i < length; i++) 
	{
     //проверяет аргумент,является ли он десятичной цифрой
	 if (isdigit(s1[i]))  s += s1[i];
	  else
		   if (isalpha(s1[i])) //параметр проверяет, является ли аргумент строчной/прописной буквой алфавита
			{
			  s += tolower(s1[i]);//преобразование прописных букв в строчные
			}
	}
	int len = s.length(); //длина строки
